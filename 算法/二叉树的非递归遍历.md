## preOrder

```py
def preOrder(tree):
    p = tree
    arr = []
    while p or arr:
        while p:
            print(p.val)
            arr.append(p)
            p = p.left
        p = arr.pop().right
```

## inOrder

```py
def inOrder(tree):
    p = tree
    arr = []
    while p or arr:
        while p:
            arr.append(p)
            p = p.left
        p = arr.pop()
        print(p.val)
        p = p.right
```

## postOrder

```py
def postOrder(tree):
    p = tree
    arr = []
    lastNode = None
    while p or arr:
        while p:
            arr.append(p)
            p = p.left
        p = arr[-1]
        if (not p.right) or lastNode == p.right:
            print(p.val)
            arr.pop()
            lastNode = p
            p = None
        else:
            p = p.right
```

## level(层次)

```py
from collections import deque
def level(tree):
    nodes = deque([tree])
    while (nodes):
        node = nodes.popleft()
        if not node: continue
        print(node.val)
        nodes.append(node.left)
        nodes.append(node.right)
```

## example

```py
class Node:
    def __init__(self, data, lr=None, rr=None):
        self.val = data
        self.left = lr
        self.right = rr

tree = Node(1)
tree.left = Node(2)
tree.right = Node(3)
tree.left.left = Node(4)
tree.left.right = Node(5)
tree.left.left.left = Node(6)
```
